class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();

        for (int i = 0; i < nums.length; i++) {
            Integer value1 = map.get(target - nums[i]);
            if (value1 != null) {
                return new int[]{value1, i};
            }
            map.put(nums[i],i);
        }
        return null;

    }
}
